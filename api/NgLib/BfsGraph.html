def add_edge (u : Int, v : Int, directed : Bool = false ) #

辺 $(u, v)$ を追加します。

directed が true の場合、 有向グラフとみなして、$u$ から $v$ への辺のみ生やします。 

graph = BfsGraph.new(n)
graph.add_edge(u, v)                 # => (u) <---w---> (v)
graph.add_edge(u, v, directed: true) # => (u) ----w---> (v)
